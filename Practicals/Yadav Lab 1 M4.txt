#Question: Area of Rectangle

#getting the width of rectangle from the user
print("Enter Width of Rectangle : ", end='')
width=float(input())                

#getting the height of rectangle from the user
print("Enter Height of Rectangle : ", end='')
height=float(input())               

#calculating the area 
area=round(width* height,2)         

#printing the area of rectangle
print("Area of Rectangle : ",area)
